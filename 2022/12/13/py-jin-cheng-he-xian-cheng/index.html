1.什么是进程 ?

进程: 是资源分配的最小单位,他也是操作系统进行资源分配和调度运行的基本单位.

比如我们的QQ , 微信 就是一个进程

2.什么是线程 ?

线程: 正在执行的一系列的任务:

‘QQ’===> 聊天(每一个聊天窗口都是一个线程)

一个进程可以拥有多个线程.

补充:

并行:

多核执行不同进程

并发:

多核交替(时间轮片转法)执行不同进程

并发和并行

并发指的是多线程同时向一个输入输出/执行任务,互不干扰.

并行是指多线程按照时间轮转片法按顺序逐个执行任务,相互有顺序.

互斥锁

mutex = threading.Lock() 
   
   
   
 mutex.acquire()  # 加锁的方法
   for i in range(nums):
       num += 1
mutex.release()  # 解锁方法
# 当多个线程几乎同时修改某一个共享数据的时候,我们是需要同步控制的
# 某个线程要更改共享的数据时,将资源的占有变为锁定状态(现在的这个时候,其他资源无法使用)
# 互斥锁 保证了每次只有一个线程进入写的操作,从而保证了多线程情况下面的数据正确性


import threading
from time import sleep

num = 0  # 全局共享的变量

mutex = threading.Lock()  # 创建了一个互斥锁,默认情况下是没有上锁的


def demo1(nums):
    global num

    mutex.acquire()  # 加锁的方法
    for i in range(nums):
        num += 1
    mutex.release()  # 解锁方法
    # print('子线程1的值: ', num)


def demo2(nums):
    global num

    mutex.acquire()
    for i in range(nums):
        num += 1
    mutex.release()
    # print('子线程2的值: ', num)


if __name__ == '__main__':
    t1 = threading.Thread(target=demo1, args=(1000000,))
    t2 = threading.Thread(target=demo2, args=(1000000,))

    # 启动线程
    t1.start()
    t1.join()
    t2.start()
    t2.join()

    print('主线程的值: ', num)

# 互斥锁 只可以锁一次,然后再解锁,否则就会导致程序阻塞

由互斥锁带来的死锁问题

# 死锁:
# 在线程之间共享多个资源的时候,如果两个线程分别占有一部分资源,并且同时等待对方的资源,这种情况下就会造成死锁

import threading
import time

# 创建一个全局锁
mutexA = threading.Lock()
mutexB = threading.Lock()


class MyThread(threading.Thread):
    def run(self):
        mutexA.acquire()  # 对mutexA上锁

        print(self.name + 'mutexA枷锁')
        time.sleep(1)

        mutexB.acquire()  # mutexB上锁
        print(self.name + 'mutexB枷锁')

        mutexB.release()

        mutexA.release()


class MyThread2(threading.Thread):
    def run(self):
        mutexB.acquire()  # 对mutexB枷锁
        print('MyThread2的 mutexB 枷锁')
        time.sleep(1)

        mutexA.acquire()  # 对mutexA枷锁
        print('MyThread2 的 mutexA的枷锁')

        mutexA.release()
        mutexB.release()


if __name__ == '__main__':
    t1 = MyThread()
    t2 = MyThread2()

    # 启动线程
    t1.start()
    t2.start()
'''
class1是:
在锁1之后没有释放就又锁2,然后释放2,再释放1
class2是:
在锁2之后没有释放就又锁1,然后释放1,再释放2


造成的是:当执行时锁1,2互相在其内部锁住对方
就像两个人对枪,互相拿着对方的人质
'''

死锁的解决方法:

import threading
import time

num = 0
mutex = threading.RLock()  # 创建一个可重入的锁 ( 可以锁多次 )


def demo1(nums):
    global num

    mutex.acquire()  # 枷锁方法
    print(1)
    mutex.acquire()
    print(2)
    for i in range(nums):
        num += 1
    mutex.release()
    print(3)
    mutex.release()
    print('子线程1: ', num)


def demo2(nums):
    global num
    mutex.acquire()
    for i in range(nums):
        num += 1

    # 解锁操作
    mutex.release()
    print('子线程2: ', num)


def main():
    t1 = threading.Thread(target=demo1, args=(1000000,))
    t2 = threading.Thread(target=demo2, args=(1000000,))

    t1.start()
    t2.start()

    print('主线程的值: ', num)


if __name__ == '__main__':
    main()

            pass

            pass

import requests
from lxml import etree
import requests
from urllib import request
import os
from queue import Queue
import threading


# 创建线程
class Producer(threading.Thread):
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/97.0.4692.99 Safari/537.36'
    }

    def __init__(self, page_queue, img_queue):
        super().__init__()
        self.page_queue = page_queue
        self.img_queue = img_queue

    def run(self):
        while True:
            if self.page_queue.empty():
                break
            url = self.page_queue.get()
            self.parse_page(url)

    # 每一页数据的解析
    def parse_page(self, url):
        response = requests.get(url=url, headers=self.headers)
        text = response.text
        html = etree.HTML(text)
        imgs = html.xpath('//img[@class="img-responsive lazy image_dta"]')
        print(len(imgs))

        # 判断是否有该文件夹
        if not os.path.exists('images'):
            os.mkdir('images')

        for img in imgs:
            img_url = img.get('data-original')
            img_name = img_url.split('/')[-1]
            try:
                img_name = img_name.replace('null', 'jpg')
            except:
                pass
            # 下载图片
            self.img_queue.put((img_url, img_name))  # 把参数以元组的形式传出去


class Consumer(threading.Thread):
    def __init__(self, page_queue, img_queue):
        super().__init__()
        self.page_queue = page_queue
        self.img_queue = img_queue

    def run(self):
        while True:
            if self.page_queue.empty() and self.img_queue.empty():
                break
            img = self.img_queue.get()
            img_url, img_name = img  # 解包
            request.urlretrieve(img_url, 'images/{}'.format(img_name))
            print('正在下载: ', img_url)


def main():
    # 创建队列(队列有200个线程)
    page_queue = Queue(200)
    img_queue = Queue(500)

    for page in range(1, 100):
        url = 'https://www.pkdoutu.com/photo/list/?page={}'.format(page)
        page_queue.put(url)

    for i in range(5):  # 创建5个线程
        t = Producer(page_queue, img_queue)
        t.start()

    for i in range(5):
        t = Consumer(page_queue, img_queue)
        t.start()


if __name__ == '__main__':
    main()
